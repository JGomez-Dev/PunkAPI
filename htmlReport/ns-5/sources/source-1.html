 package com.jgomez.punkapi.data.datasource.local.di
 
 import android.content.Context
 import androidx.room.Room
 import dagger.Module
 import dagger.Provides
 import dagger.hilt.InstallIn
 import dagger.hilt.android.qualifiers.ApplicationContext
 import dagger.hilt.components.SingletonComponent
 import com.jgomez.punkapi.data.datasource.local.database.AppDatabase
 import com.jgomez.punkapi.data.datasource.local.database.HomeDao
 import javax.inject.Singleton
 
 
 @Module
 @InstallIn(SingletonComponent::class)
 class DatabaseModule {
     @Provides
     fun provideHomeDao(appDatabase: AppDatabase): HomeDao {
         return appDatabase.homeDao()
     }
 
     @Provides
     @Singleton
     fun provideAppDatabase(@ApplicationContext appContext: Context): AppDatabase {
         return Room.databaseBuilder(
             appContext,
             AppDatabase::class.java,
             "Home"
         ).build()
     }
 }
